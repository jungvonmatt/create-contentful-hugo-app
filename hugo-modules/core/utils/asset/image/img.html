{{/* utils/asset/image Render Contentful asset (image/...). "//") (print "https:" .) .) -}} {{/* Add preload hint here as we have no srcset available */}} {{- $preload_attr := (dict "as" "image" "href" $src "type" $mime_type "media" $media ) -}} {{- if and $preload (not (in ($globals.Scratch.Get "preload") $preload_attr)) -}} {{- $globals.Scratch.Add "preload" (slice $preload_attr) -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}}
